

create table recruit(
r_num number(8) not null,
r_id varchar2(30) not null,
r_image blob not null,
r_title varchar2(50) not null,
r_start_date varchar2(20) not null,
r_end_date varchar2(20) not null,
r_people number(30) not null,
r_people_count number(30) default 0 not null,
r_content clob not null,
r_filename varchar2(30) not null,
constraint recruit_pk primary key(r_num),
constraint recruit_fk foreign key(r_id) references spmember(id)
);
create sequence recruit_seq;


create table volunteer (
v_num number(8) not null,
v_id varchar2(30) not null,
v_date varchar2(30) not null,
v_ip varchar2(40) not null,
r_num number(8) not null,
v_status number(5) not null,
constraint volunteer_pk primary key(v_num),
constraint volunteer_fk1 foreign key (v_id) references spmember (id),
constraint volunteer_fk2 foreign key (r_num) references recruit (r_num)
);
create sequence volunteer_seq;
